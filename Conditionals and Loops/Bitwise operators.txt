#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
//Complete the following function.


void calculate_the_maximum(int n, int k) {
int m_and=0,m_or=0,m_xor=0;
for (int i=1;i<=n;i++)
{ for (int j=i+1;j<=n;j++)
    {if (((i&j) > m_and) && ((i&j) < k))
      { m_and=i&j; }
    if (((i|j) > m_or) && ((i|j) < k))
      {m_or=i|j; }
    if (((i^j) > m_xor) && ((i^j) < k))
      {m_xor=i^j; }
    }
}
printf("%d\n%d\n%d\n",m_and,m_or,m_xor);
}

int main() {
    int n, k;
  
    scanf("%d %d", &n, &k);
    calculate_the_maximum(n, k);
 
    return 0;
}
